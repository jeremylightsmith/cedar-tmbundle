## install

{{
  cd ~/Library/Application\ Support/TextMate/Bundles/
  git clone git://github.com/jeremylightsmith/cedar-tmbundle.git Cedar.tmbundle
  osascript -e 'tell app "TextMate" to reload bundles'
}}
